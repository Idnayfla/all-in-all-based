## Task 2: Welcome audio synthesis

**Files:**
- Create: `lib/welcomeAudio.ts`

Synthesizes a three-layer dramatic sound using the Web Audio API. No audio file — generates sound programmatically. Total duration ~1.2s.

- [ ] **Step 1: Create `lib/welcomeAudio.ts`**

```typescript
export function playWelcomeAudio(): void {
  try {
    const ctx = new AudioContext()
    const master = ctx.createGain()
    master.gain.value = 0.8
    master.connect(ctx.destination)

    const now = ctx.currentTime

    // Layer 1: Deep bass hit — 60Hz sine, decays over 600ms
    const bass = ctx.createOscillator()
    const bassGain = ctx.createGain()
    bass.type = 'sine'
    bass.frequency.value = 60
    bassGain.gain.setValueAtTime(0.8, now)
    bassGain.gain.exponentialRampToValueAtTime(0.001, now + 0.6)
    bass.connect(bassGain)
    bassGain.connect(master)
    bass.start(now)
    bass.stop(now + 0.6)

    // Layer 2: Rising sweep — 200Hz → 900Hz over 800ms
    const sweep = ctx.createOscillator()
    const sweepGain = ctx.createGain()
    sweep.type = 'sine'
    sweep.frequency.setValueAtTime(200, now)
    sweep.frequency.exponentialRampToValueAtTime(900, now + 0.8)
    sweepGain.gain.setValueAtTime(0, now)
    sweepGain.gain.linearRampToValueAtTime(0.3, now + 0.1)
    sweepGain.gain.linearRampToValueAtTime(0, now + 0.8)
    sweep.connect(sweepGain)
    sweepGain.connect(master)
    sweep.start(now)
    sweep.stop(now + 0.8)

    // Layer 3a: A4 chord shimmer — 440Hz triangle
    const chord1 = ctx.createOscillator()
    const chord1Gain = ctx.createGain()
    chord1.type = 'triangle'
    chord1.frequency.value = 440
    chord1Gain.gain.setValueAtTime(0, now)
    chord1Gain.gain.linearRampToValueAtTime(0.15, now + 0.2)
    chord1Gain.gain.linearRampToValueAtTime(0, now + 1.2)
    chord1.connect(chord1Gain)
    chord1Gain.connect(master)
    chord1.start(now)
    chord1.stop(now + 1.2)

    // Layer 3b: C#5 chord shimmer — 554Hz triangle
    const chord2 = ctx.createOscillator()
    const chord2Gain = ctx.createGain()
    chord2.type = 'triangle'
    chord2.frequency.value = 554
    chord2Gain.gain.setValueAtTime(0, now)
    chord2Gain.gain.linearRampToValueAtTime(0.15, now + 0.2)
    chord2Gain.gain.linearRampToValueAtTime(0, now + 1.2)
    chord2.connect(chord2Gain)
    chord2Gain.connect(master)
    chord2.start(now)
    chord2.stop(now + 1.2)

    // Close context after all sounds finish
    setTimeout(() => ctx.close().catch(() => {}), 1600)
  } catch {
    // AudioContext unsupported or blocked — silent fallback, splash fades normally
  }
}
```

```tsx
'use client'

import { useState, useEffect } from 'react'
import { motion, AnimatePresence } from 'framer-motion'
import { playWelcomeAudio } from '@/lib/welcomeAudio'

type State = 'waiting' | 'exiting' | 'done'

export default function LaunchSplash() {
  const [state, setState] = useState<State | null>(null)

  useEffect(() => {
    const isStandalone =
      window.matchMedia('(display-mode: standalone)').matches ||
      ('standalone' in navigator && (navigator as any).standalone === true)

    if (isStandalone) setState('waiting')
  }, [])

  function handleTap() {
    if (state !== 'waiting') return
    playWelcomeAudio()
    setState('exiting')
    setTimeout(() => setState('done'), 450)
  }

  if (state === null || state === 'done') return null

  return (
    <AnimatePresence>
      {state !== 'done' && (
        <motion.div
          key="launch-splash"
          className="launch-splash"
          onClick={handleTap}
          initial={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          transition={{ duration: 0.4, ease: 'easeOut' }}
        >
          <div className="launch-splash__pulse" />
          <div className="launch-splash__content">
            <h1 className="launch-splash__title">Based</h1>
            <p className="launch-splash__tagline">Your personal AI studio</p>
          </div>
          <motion.p
            className="launch-splash__hint"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 0.8, duration: 0.4 }}
          >
            Tap anywhere to enter
          </motion.p>
        </motion.div>
      )}
    </AnimatePresence>
  )
}
```

```tsx
'use client'

import { useState, useEffect, useRef } from 'react'
import { motion, AnimatePresence } from 'framer-motion'

type Platform = 'android' | 'ios'

export default function InstallPrompt() {
  const [platform, setPlatform] = useState<Platform | null>(null)
  const deferredPrompt = useRef<any>(null)

  useEffect(() => {
    // Never show if already installed
    const isStandalone =
      window.matchMedia('(display-mode: standalone)').matches ||
      ('standalone' in navigator && (navigator as any).standalone === true)
    if (isStandalone) return

    // Never show if dismissed this session
    if (sessionStorage.getItem('install-dismissed')) return

    const ua = navigator.userAgent
    const isIOS = /iphone|ipad|ipod/i.test(ua) && !('MSStream' in window)

    if (isIOS) {
      setTimeout(() => setPlatform('ios'), 3000)
      return
    }

    function onBeforeInstall(e: Event) {
      e.preventDefault()
      deferredPrompt.current = e
      setTimeout(() => setPlatform('android'), 3000)
    }

    window.addEventListener('beforeinstallprompt', onBeforeInstall)
    return () => window.removeEventListener('beforeinstallprompt', onBeforeInstall)
  }, [])

  async function handleInstall() {
    if (!deferredPrompt.current) return
    deferredPrompt.current.prompt()
    const { outcome } = await deferredPrompt.current.userChoice
    deferredPrompt.current = null
    if (outcome === 'accepted') setPlatform(null)
  }

  function handleDismiss() {
    sessionStorage.setItem('install-dismissed', '1')
    setPlatform(null)
  }

  return (
    <AnimatePresence>
      {platform && (
        <motion.div
          key="install-prompt"
          className="install-prompt"
          initial={{ y: 80, opacity: 0 }}
          animate={{ y: 0, opacity: 1 }}
          exit={{ y: 80, opacity: 0 }}
          transition={{ type: 'spring', stiffness: 280, damping: 28 }}
        >
          {platform === 'android' ? (
            <>
              <p className="install-prompt__text">Install Based for the full experience</p>
              <button className="install-prompt__btn" onClick={handleInstall}>Install</button>
            </>
          ) : (
            <p className="install-prompt__text">
              Install Based — tap <strong>Share</strong> then <strong>Add to Home Screen</strong>
            </p>
          )}
          <button className="install-prompt__dismiss" onClick={handleDismiss} aria-label="Dismiss">×</button>
        </motion.div>
      )}
    </AnimatePresence>
  )
}
```

    const SIZES = [
      { w: 750,  h: 1334 },
      { w: 1179, h: 2556 },
      { w: 1284, h: 2778 },
      { w: 1290, h: 2796 },
      { w: 1640, h: 2360 },
      { w: 1668, h: 2388 },

    function draw(w, h) {
      const canvas = document.createElement('canvas')
      canvas.width = w
      canvas.height = h
      const ctx = canvas.getContext('2d')

      // Background
      ctx.fillStyle = '#0a0a0f'

      ctx.fillStyle = 'rgba(255,255,255,0.38)'

          log.textContent += `✓ splash-${w}x${h}.png\n`
        }, 'image/png')
      })
    }

```tsx
'use client'

import { useEffect, useState } from 'react'
import { motion, AnimatePresence } from 'framer-motion'

export default function ServiceWorkerInit() {
  const [updateReady, setUpdateReady] = useState(false)

  useEffect(() => {
    if (!('serviceWorker' in navigator)) return

    navigator.serviceWorker.register('/sw.js').catch(() => {})

    function onControllerChange() {
      setUpdateReady(true)
    }

    navigator.serviceWorker.addEventListener('controllerchange', onControllerChange)
    return () => navigator.serviceWorker.removeEventListener('controllerchange', onControllerChange)
  }, [])

  useEffect(() => {
    if (!updateReady) return
    const timer = setTimeout(() => setUpdateReady(false), 10_000)
    return () => clearTimeout(timer)
  }, [updateReady])

  return (
    <AnimatePresence>
      {updateReady && (
        <motion.div
          key="sw-update"
          className="sw-update-toast"
          initial={{ y: 80, opacity: 0 }}
          animate={{ y: 0, opacity: 1 }}
          exit={{ y: 80, opacity: 0 }}
          transition={{ type: 'spring', stiffness: 280, damping: 28 }}
        >
          <span>Based updated</span>
          <button
            className="sw-update-toast__reload"
            onClick={() => window.location.reload()}
          >
            Reload
          </button>
        </motion.div>
      )}
    </AnimatePresence>
  )
}
```

## Task 7: Maskable icon

**Files:**
- Create: `public/icon-512-maskable.png` (generated)
- Modify: `public/manifest.json`

Android adaptive icons crop to a circle/squircle shape. The current `icon-512.png` has no padding — the logo gets clipped. This task generates a version with the logo centered in the inner 72% of the canvas.

- [ ] **Step 1: Generate the maskable icon**

Create `scripts/generate-maskable.html` (anywhere on your machine — not in the repo):

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Based — Maskable Icon Generator</title>
  <style>
    body { font-family: sans-serif; padding: 32px; background: #111; color: #eee; }
    button { background: #7c6af7; color: #fff; border: none; padding: 12px 24px; border-radius: 8px; font-size: 16px; cursor: pointer; }
    canvas { display: block; margin: 16px 0; border-radius: 8px; }
    #status { margin-top: 12px; font-family: monospace; color: #aaa; }
  </style>
</head>
<body>
  <h2>Based Maskable Icon Generator</h2>
  <p>Drop your <code>icon-512.png</code> below, preview the result, then download.</p>
  <input type="file" id="file" accept="image/png">
  <canvas id="preview" width="512" height="512"></canvas>
  <button id="download" disabled>Download icon-512-maskable.png</button>
  <p id="status">Load an icon to begin.</p>
  <script>
    const canvas = document.getElementById('preview')
    const ctx = canvas.getContext('2d')

    document.getElementById('file').onchange = (e) => {
      const file = e.target.files[0]
      if (!file) return
      const img = new Image()
      img.onload = () => {
        // Background fill
        ctx.fillStyle = '#0a0a0f'
        ctx.fillRect(0, 0, 512, 512)

        // Draw icon scaled to 72% centered (safe zone for maskable)
        const safeSize = 512 * 0.72
        const offset = (512 - safeSize) / 2
        ctx.drawImage(img, offset, offset, safeSize, safeSize)

        document.getElementById('download').disabled = false
        document.getElementById('status').textContent = '✓ Preview ready — check the safe zone padding, then download.'
      }
      img.src = URL.createObjectURL(file)
    }

    document.getElementById('download').onclick = () => {
      canvas.toBlob(blob => {
        const a = document.createElement('a')
        a.href = URL.createObjectURL(blob)
        a.download = 'icon-512-maskable.png'
        a.click()
        document.getElementById('status').textContent = '✓ Downloaded icon-512-maskable.png — move it to public/'
      }, 'image/png')
    }
  </script>
</body>
</html>
```
